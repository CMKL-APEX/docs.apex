## วิธีที่ 1: Secure Shell (SSH)

ผู้ใช้งานหลากหลายท่านโดยส่วนใหญ่ใช้ command line interface (CLI) เพื่อที่จะเขื่อมต่อไปยัง APEX cluster
SSH clients ที่แนะนำได้แก่:

- Terminal , Termius (Linux, MacOS)
- MobaXterm, PuTTY, PowerShell (Windows)

ขั้นตอนแรก คุณตจำเป็นต้องเชื่อมต่อไปยัง APEX cluster ผ่านคำสั่ง SSH:

export const cli = `ssh <username>@apex-login.cmkl.ac.th`;
export const cli_out = `ssh <username>@archon-2:~$`;
export const fileName = "Terminal";
export const highlights = ["<username>"];
export const highlights2 = ["archon-2"];

<Code code={cli} lang="bash" title={fileName} mark={highlights} />
